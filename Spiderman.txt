  <script>
    const btnNo = document.getElementById('btn-no');
    const tarjeta = document.getElementById('tarjeta-invitacion');
    const pantallaBloqueo = document.getElementById('bloqueo-audio');
    const musica = document.getElementById('musica-toy');
    const fuenteAudio = document.getElementById('fuente-audio');
    let musicaIniciada = false;

    // Esta función se ejecuta con el primer toque real del usuario, rompiendo el bloqueo del celular
    function entrarInvitacion() {
      if (!musicaIniciada) {
        // Rompe-caché dinámico
        fuenteAudio.src = "Yo_Soy_Tu_Amigo_Fiel.mp3?v=" + new Date().getTime();
        musica.load();
        
        musica.play()
          .then(() => { musicaIniciada = true; })
          .catch((e) => console.log("Audio bloqueado:", e));
      }

      // Oculta la pantalla de bienvenida y muestra la tarjeta animada
      pantallaBloqueo.style.opacity = "0";
      pantallaBloqueo.style.visibility = "hidden";
      tarjeta.classList.add("show");
    }

    // Comportamiento del botón "No" (Mover arriba, derecha y abajo sin ir a la izquierda)
    function moverBotonNo() {
      // Valores positivos en X obligan al botón a ir a la derecha (evita colisionar con el botón Sí)
      const randomX = Math.floor(Math.random() * 120) + 40; 
      
      // Valores en Y que permiten subir (negativos) o bajar (positivos)
      const randomY = Math.floor(Math.random() * 180) - 90; 

      btnNo.style.transform = `translate(${randomX}px, ${randomY}px)`;
    }

    btnNo.addEventListener('mouseenter', moverBotonNo);

    btnNo.addEventListener('touchstart', (e) => {
      e.preventDefault();
      moverBotonNo();
    });

    tarjeta.addEventListener('mouseleave', () => {
      btnNo.style.transform = 'translate(0px, 0px)';
    });

    function abrirWhatsApp() {
      const numero = "50661547634";
      const mensaje = encodeURIComponent("¡Sí, quiero ir contigo al cine a ver Toy Story 5! 🚀🤠🎬");
      
      const deepLink = `whatsapp://send?phone=${numero}&text=${mensaje}`;
      const webLink = `https://whatsapp.com{numero}&text=${mensaje}`;
      
      window.location.href = deepLink;
      
      setTimeout(() => {
        window.open(webLink, '_blank');
      }, 250);
    }
  </script>
